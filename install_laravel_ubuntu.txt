1. Cài đặt apache
  $ sudo apt-get update
  $ sudo apt-get install apache2

2. Cài đặt MySQL

  $ sudo apt-get update

  $ sudo apt-get install mysql-server

3. Cài đặt server side script

  $ sudo apt-get update
  $ sudo apt-get install python-software-properties
  $ sudo add-apt-repository ppa:ondrej/php
        |-> if doesn't work : sudo apt-get install software-properties-common
  $ sudo apt-get update
  $ sudo apt-get install php -> sudo apt-get install php7.3
  $ sudo apt-get install php-pear php-curl php-dev php-gd php-mbstring php-zip php-mysql php-xml ----> apt install php-pear php7.3-curl php7.3-dev php7.3-gd php7.3-mbstring php7.3-zip php7.3-mysql php7.3-xml php7.3-fpm libapache2-mod-php7.3 php7.3-imagick php7.3-recode php7.3-tidy php7.3-xmlrpc php7.3-intl
  
4. Change versiondefault;
  $ update-alternatives --set php /usr/bin/phpxx.x
  `update-alternatives --set php /usr/bin/php7.3`
  $ sudo apache2 restart

  - For Debian:
    sudo apt install ca-certificates apt-transport-https
    wget -q https://packages.sury.org/php/apt.gpg -O- | sudo apt-key add -
    echo "deb https://packages.sury.org/php/ stretch main" | sudo tee /etc/apt/sources.list.d/php.list
    cd /etc/apt/sources.list.d/
    sudo mv ondrej-ubuntu* /tmp
    sudo apt -y update
    sudo apt install php7.3


5. Cài đặt composer

  $ sudo apt-get update

  $ sudo apt-get install curl php-cli php-mbstring git unzip

  $ cd ~

  $ curl -sS https://getcomposer.org/installer -o composer-setup.php

  $ sudo php composer-setup.php --install-dir=/usr/local/bin --filename=composer


6. Cài đặt phpmyadmin

  $ sudo apt-get install phpmyadmin

  Nếu sau khi cài đặt không chạy

  $ sudo ln -s /etc/phpmyadmin/apache.conf /etc/apache2/conf-available/phpmyadmin.conf
  $ sudo a2enconf phpmyadmin

  $ sudo service apache2 reload
